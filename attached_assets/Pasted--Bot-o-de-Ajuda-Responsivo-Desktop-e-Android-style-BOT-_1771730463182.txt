<!-- Script para abrir o chat -->
<script>
  function abrirChatAjuda() {
    // Tenta abrir o widget RelpFlow
    if (typeof RelpFlow !== 'undefined' && typeof RelpFlow.open === 'function') {
      RelpFlow.open();
    } else if (typeof window.RelpFlowWidget !== 'undefined' && typeof window.RelpFlowWidget.open === 'function') {
      window.RelpFlowWidget.open();
    } else {
      // Fallback: evento personalizado
      window.dispatchEvent(new CustomEvent('relpflow:open'));
    }
    
    // Ajusta posicionamento após abrir (garante que não cubra menus)
    setTimeout(ajustarChatMobile, 300);
  }

  function ajustarChatMobile() {
    if (window.innerWidth <= 768) {
      const chatWidget = document.querySelector('.relpflow-widget, .relpflow-container, [data-relpflow] iframe');
      if (chatWidget) {
        // Garante margens seguras para menus
        chatWidget.style.maxHeight = 'calc(100vh - 110px)';
        chatWidget.style.bottom = '55px';
        chatWidget.style.top = '60px';
      }
    }
  }

  // Ajusta ao redimensionar a tela
  window.addEventListener('resize', function() {
    if (window.innerWidth <= 768) {
      ajustarChatMobile();
    }
  });

  // Ajusta após carregar a página
  window.addEventListener('load', function() {
    setTimeout(ajustarChatMobile, 500);
  });
</script>
